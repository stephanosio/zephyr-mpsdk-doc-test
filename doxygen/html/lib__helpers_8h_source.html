17#define read_sysreg(reg) \
18({ \
19 uint64_t val; \
20 __asm__ volatile ("mrs %0, " STRINGIFY(reg) \
21 : "=r" (val) :: "memory"); \
22 val; \
23})

25#define write_sysreg(val, reg) \

29})

31#define zero_sysreg(reg) \
32({ \

163static inline bool is_el_implemented(unsigned int el)
164{
165 unsigned int shift;
166
167 if (el > 3) {
168 return false;
169 }
170
171 shift = ID_AA64PFR0_EL1_SHIFT * el;
172
173 return (((read_id_aa64pfr0_el1() >> shift) & ID_AA64PFR0_ELX_MASK) != 0U);
174}
175
176static inline bool is_el_highest_implemented(void)
177{
178 uint32_t el_highest;
179 uint32_t curr_el;
180
181 el_highest = read_id_aa64pfr0_el1() & 0xFFFF;
182 el_highest = (31U - __builtin_clz(el_highest)) / 4;
183
184 curr_el = GET_EL(read_currentel());
185
186 if (curr_el < el_highest)
187 return false;
188
189 return true;
190}
