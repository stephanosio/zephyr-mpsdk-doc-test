24#define STM32_REMAP_REG_MASK 0x1U
25#define STM32_REMAP_REG_SHIFT 0U
26#define STM32_REMAP_SHIFT_MASK 0x1FU
27#define STM32_REMAP_SHIFT_SHIFT 1U
28#define STM32_REMAP_MASK_MASK 0x3U
29#define STM32_REMAP_MASK_SHIFT 6U
30#define STM32_REMAP_VAL_MASK 0x3U
31#define STM32_REMAP_VAL_SHIFT 8U
32
33#define STM32_REMAP(val, mask, shift, reg) \
34 ((((reg) & STM32_REMAP_REG_MASK) << STM32_REMAP_REG_SHIFT) | \
35 (((shift) & STM32_REMAP_SHIFT_MASK) << STM32_REMAP_SHIFT_SHIFT) | \
36 (((mask) & STM32_REMAP_MASK_MASK) << STM32_REMAP_MASK_SHIFT) | \
37 (((val) & STM32_REMAP_VAL_MASK) << STM32_REMAP_VAL_SHIFT))
38
