23{
24 __asm__ volatile("movq %0, %1"
25 :
26 : "r"(data), "m" (*(volatile uint64_t *)
27 (uintptr_t) addr)
28 : "memory");
29}

31static ALWAYS_INLINE uint64_t sys_read64(mm_reg_t addr)
32{
33 uint64_t ret;
34
35 __asm__ volatile("movq %1, %0"
36 : "=r"(ret)
37 : "m" (*(volatile uint64_t *)(uintptr_t) addr)
38 : "memory");
39
40 return ret;
41}
