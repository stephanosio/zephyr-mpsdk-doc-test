216{
217 char type[10];

219 switch (addr->type) {

222 break;
223 case BT_ADDR_LE_RANDOM:
224 strcpy(type, "random");
225 break;
226 case BT_ADDR_LE_PUBLIC_ID:
227 strcpy(type, "public-id");
228 break;
229 case BT_ADDR_LE_RANDOM_ID:
230 strcpy(type, "random-id");
231 break;
232 default:
233 snprintk(type, sizeof(type), "0x%02x", addr->type);
234 break;
235 }
236
237 return snprintk(str, len, "%02X:%02X:%02X:%02X:%02X:%02X (%s)",
238 addr->a.val[5], addr->a.val[4], addr->a.val[3],
239 addr->a.val[2], addr->a.val[1], addr->a.val[0], type);
240}
