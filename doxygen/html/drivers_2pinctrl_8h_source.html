339static inline int pinctrl_apply_state(const struct pinctrl_dev_config *config,
340 uint8_t id)
341{
342 int ret;
343 const struct pinctrl_state *state;
344
345 ret = pinctrl_lookup_state(config, id, &state);
346 if (ret < 0) {
347 return ret;
348 }
349
350 return pinctrl_apply_state_direct(config, state);
351}
