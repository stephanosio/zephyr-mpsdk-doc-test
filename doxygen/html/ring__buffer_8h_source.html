144static inline void ring_buf_init(struct ring_buf *buf,
145 uint32_t size,
146 uint8_t *data)
147{
148 __ASSERT(size < RING_BUFFER_MAX_SIZE, RING_BUFFER_SIZE_ASSERT_MSG);

150 buf->size = size;
