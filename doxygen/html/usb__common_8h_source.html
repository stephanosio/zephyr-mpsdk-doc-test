100/* bmAttributes:
101 * D7:Reserved, always 1,
102 * D6:Self-Powered -> 1,
103 * D5:Remote Wakeup -> 0,
104 * D4...0:Reserved -> 0
105 */
106#define USB_CONFIGURATION_ATTRIBUTES_REMOTE_WAKEUP __DEPRECATED_MACRO BIT(5)
107#define USB_CONFIGURATION_ATTRIBUTES_SELF_POWERED __DEPRECATED_MACRO BIT(6)
108#define USB_CONFIGURATION_ATTRIBUTES BIT(7) __DEPRECATED_MACRO \
109 | ((COND_CODE_1(CONFIG_USB_SELF_POWERED, \
110 (USB_CONFIGURATION_ATTRIBUTES_SELF_POWERED), (0))) \
111 | (COND_CODE_1(CONFIG_USB_DEVICE_REMOTE_WAKEUP, \
112 (USB_CONFIGURATION_ATTRIBUTES_REMOTE_WAKEUP), (0))))
