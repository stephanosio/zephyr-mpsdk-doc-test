761static inline int can_set_bitrate(const struct device *dev,
762 uint32_t bitrate,
763 uint32_t bitrate_data)
764{
765 struct can_timing timing;
766#ifdef CONFIG_CAN_FD_MODE
767 struct can_timing timing_data;
768#endif
769 int ret;
770
771 ret = can_calc_timing(dev, &timing, bitrate, 875);
772 if (ret < 0) {
773 return -EINVAL;
774 }
775
776 timing.sjw = CAN_SJW_NO_CHANGE;
777
778#ifdef CONFIG_CAN_FD_MODE
779 ret = can_calc_timing_data(dev, &timing_data, bitrate_data, 875);
780 if (ret < 0) {
781 return -EINVAL;
782 }
783
784 timing_data.sjw = CAN_SJW_NO_CHANGE;
785
786 return can_set_timing(dev, &timing, &timing_data);
787#else /* CONFIG_CAN_FD_MODE */
788 return can_set_timing(dev, &timing, NULL);
789#endif /* !CONFIG_CAN_FD_MODE */
790}

1172static inline void can_copy_frame_to_zframe(const struct can_frame *frame,
1173 struct zcan_frame *zframe)
1174{
1175 zframe->id_type = (frame->can_id & BIT(31)) >> 31;
1176 zframe->rtr = (frame->can_id & BIT(30)) >> 30;
1177 zframe->id = frame->can_id & BIT_MASK(29);
1178 zframe->dlc = frame->can_dlc;
1179 memcpy(zframe->data, frame->data, sizeof(zframe->data));
1180}

1188static inline void can_copy_zframe_to_frame(const struct zcan_frame *zframe,
1189 struct can_frame *frame)
1190{
1191 frame->can_id = (zframe->id_type << 31) | (zframe->rtr << 30) | zframe->id;
1192 frame->can_dlc = zframe->dlc;
1193 memcpy(frame->data, zframe->data, sizeof(frame->data));
1194}

1202static inline void can_copy_filter_to_zfilter(const struct can_filter *filter,
1203 struct zcan_filter *zfilter)
1204{
1205 zfilter->id_type = (filter->can_id & BIT(31)) >> 31;
1206 zfilter->rtr = (filter->can_id & BIT(30)) >> 30;
1207 zfilter->id = filter->can_id & BIT_MASK(29);
1208 zfilter->rtr_mask = (filter->can_mask & BIT(30)) >> 30;
1209 zfilter->id_mask = filter->can_mask & BIT_MASK(29);

1218static inline void can_copy_zfilter_to_filter(const struct zcan_filter *zfilter,
1219 struct can_filter *filter)
1220{
1221 filter->can_id = (zfilter->id_type << 31) |
1222 (zfilter->rtr << 30) | zfilter->id;
1223 filter->can_mask = (zfilter->rtr_mask << 30) |
1224 (zfilter->id_type << 31) | zfilter->id_mask;
