270__syscall uint32_t counter_us_to_ticks(const struct device *dev, uint64_t us);
271
272static inline uint32_t z_impl_counter_us_to_ticks(const struct device *dev,
273 uint64_t us)
274{
275 uint64_t ticks = (us * z_impl_counter_get_frequency(dev)) / USEC_PER_SEC;
276
277 return (ticks > (uint64_t)UINT32_MAX) ? UINT32_MAX : ticks;
278}

392static inline int z_impl_counter_set_channel_alarm(const struct device *dev,
