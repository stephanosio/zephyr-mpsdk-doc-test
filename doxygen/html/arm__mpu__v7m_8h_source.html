206/*
207 * @brief Evaluate Write-ability
208 *
209 * Evaluate whether the access permissions include write-ability.
210 *
211 * @param attr The k_mem_partition_attr_t object holding the
212 * MPU attributes to be checked against write-ability.
213 */
214#define K_MEM_PARTITION_IS_WRITABLE(attr) \
215 ({ \
216 int __is_writable__; \
217 switch (attr.rasr_attr & MPU_RASR_AP_Msk) { \
218 case P_RW_U_RW_Msk: \
219 case P_RW_U_RO_Msk: \
220 case P_RW_U_NA_Msk: \
221 __is_writable__ = 1; \
222 break; \
223 default: \
224 __is_writable__ = 0; \
225 } \
226 __is_writable__; \
227 })
