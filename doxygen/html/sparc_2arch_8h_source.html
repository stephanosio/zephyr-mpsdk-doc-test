63static ALWAYS_INLINE unsigned int z_sparc_set_pil_inline(unsigned int newpil)
64{
65 register uint32_t oldpil __asm__ ("o0") = newpil;
66
67 __asm__ volatile (
68 "ta %1\nnop\n" :
69 "=r" (oldpil) :
70 "i" (SPARC_SW_TRAP_SET_PIL), "r" (oldpil) :
71 "memory"
72 );
73 return oldpil;
74}
