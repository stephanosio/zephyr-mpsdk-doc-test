122/*
123 * The first word of the MSI capability is shared with the
124 * capability ID and list link. The high 16 bits are the MCR.
125 */
126
127#define PCIE_MSI_MCR 0U
128
129#define PCIE_MSI_MCR_EN 0x00010000U /* enable MSI */
130#define PCIE_MSI_MCR_MMC 0x000E0000U /* Multi Messages Capable mask */
131#define PCIE_MSI_MCR_MMC_SHIFT 17
132#define PCIE_MSI_MCR_MME 0x00700000U /* mask of # of enabled IRQs */
133#define PCIE_MSI_MCR_MME_SHIFT 20
134#define PCIE_MSI_MCR_64 0x00800000U /* 64-bit MSI */
135
136/*
137 * The MAP follows the MCR. If PCIE_MSI_MCR_64, then the MAP
138 * is two words long. The MDR follows immediately after the MAP.
139 */
140
141#define PCIE_MSI_MAP0 1U
142#define PCIE_MSI_MAP1_64 2U
143#define PCIE_MSI_MDR_32 2U
144#define PCIE_MSI_MDR_64 3U
145
146/*
147 * As for MSI, he first word of the MSI-X capability is shared
148 * with the capability ID and list link. The high 16 bits are the MCR.
149 */
150
151#define PCIE_MSIX_MCR 0U
152
153#define PCIE_MSIX_MCR_EN 0x80000000U /* Enable MSI-X */
154#define PCIE_MSIX_MCR_FMASK 0x40000000U /* Function Mask */
155#define PCIE_MSIX_MCR_TSIZE 0x07FF0000U /* Table size mask */
156#define PCIE_MSIX_MCR_TSIZE_SHIFT 16
157#define PCIE_MSIR_TABLE_ENTRY_SIZE 16
158
159#define PCIE_MSIX_TR 1U
160#define PCIE_MSIX_TR_BIR 0x00000007U /* Table BIR mask */
161#define PCIE_MSIX_TR_OFFSET 0xFFFFFFF8U /* Offset mask */

163#define PCIE_MSIX_PBA 2U
164#define PCIE_MSIX_PBA_BIR 0x00000007U /* PBA BIR mask */
165#define PCIE_MSIX_PBA_OFFSET 0xFFFFFFF8U /* Offset mask */
