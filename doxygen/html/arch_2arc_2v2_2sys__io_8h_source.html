89static ALWAYS_INLINE
90 int sys_io_test_bit(io_port_t port, unsigned int bit)
91{
92 uint32_t status = _ARC_V2_STATUS32;
93 uint32_t reg = 0;
94 uint32_t ret;
95
96 __asm__ volatile("lr %2, [%1]\n"
97 "btst %2, %3\n"
98 "lr %0, [%4];\n\t"
99 : "=r" (ret)
100 : "ir" (port),
101 "r" (reg), "ir" (bit), "i" (status)
102 : "memory", "cc");
103
104 return !(ret & _ARC_V2_STATUS32_Z);
105}
