142 * generate a symbol to mark the start of the objects array for

420__pinned_func
421static inline bool lnkr_is_pinned(uint8_t *addr)
422{
423 if ((addr >= (uint8_t *)lnkr_pinned_start) &&
424 (addr < (uint8_t *)lnkr_pinned_end)) {
425 return true;
426 } else {
427 return false;
428 }
429}
430
431__pinned_func
432static inline bool lnkr_is_region_pinned(uint8_t *addr, size_t sz)
433{
434 if ((addr >= (uint8_t *)lnkr_pinned_start) &&
435 ((addr + sz) < (uint8_t *)lnkr_pinned_end)) {
436 return true;
437 } else {
438 return false;
439 }
440}
