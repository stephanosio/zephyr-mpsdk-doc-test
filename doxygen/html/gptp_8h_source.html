76static inline double gptp_pow2(int exp)
77{
78 double res;
79
80 if (exp >= 0) {
81 res = 1 << exp;
82 } else {
83 res = 1.0;
84
85 while (exp++) {
86 res /= 2;
87 }
88 }
89
90 return res;
91}
