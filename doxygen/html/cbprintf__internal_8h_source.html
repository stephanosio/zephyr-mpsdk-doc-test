51static inline void z_cbprintf_wcpy(int *dst, int *src, size_t len)
52{
53 for (size_t i = 0; i < len; i++) {
54 dst[i] = src[i];
55 }
56}

65#if defined(__sparc__)
66/* The SPARC V8 ABI guarantees that the arguments of a variable argument
67 * list function are stored on the stack at addresses which are 32-bit
68 * aligned. It means that variables of type unit64_t and double may not
69 * be properly aligned on the stack.
70 *
71 * The compiler is aware of the ABI and takes care of this. However,
72 * as we are directly accessing the variable argument list here, we need
73 * to take the alignment into consideration and copy 64-bit arguments
74 * as 32-bit words.
75 */
76#define Z_CBPRINTF_VA_STACK_LL_DBL_MEMCPY 1
77#else

263#define Z_CBPRINTF_PACK_ARG2(_buf, _idx, _align_offset, _max, \
264 _cfg_flags, _s_idx, _s_buf, _arg) \
265do { \
266 BUILD_ASSERT(!((sizeof(double) < VA_STACK_ALIGN(long double)) && \
267 Z_CBPRINTF_IS_LONGDOUBLE(_arg) && \
268 !IS_ENABLED(CONFIG_CBPRINTF_PACKAGE_LONGDOUBLE)),\
269 "Packaging of long double not enabled in Kconfig."); \
270 while (_align_offset % Z_CBPRINTF_ALIGNMENT(_arg)) { \
271 _idx += sizeof(int); \
272 _align_offset += sizeof(int); \
273 } \
274 uint32_t _arg_size = Z_CBPRINTF_ARG_SIZE(_arg); \
275 if (Z_CBPRINTF_IS_PCHAR(_arg, 0)) { \
276 _s_buf[_s_idx++] = _idx / sizeof(int); \
277 } \
278 if (_buf && _idx < (int)_max) { \
279 Z_CBPRINTF_STORE_ARG(&_buf[_idx], _arg); \
280 } \
281 _idx += _arg_size; \
282 _align_offset += _arg_size; \
283} while (0)

340#define Z_CBPRINTF_STATIC_PACKAGE_GENERIC(buf, _inlen, _outlen, _align_offset, \
341 _flags, ... /* fmt, ... */) \
342do { \
343 _Pragma("GCC diagnostic push") \
344 _Pragma("GCC diagnostic ignored \"-Wpointer-arith\"") \
345 Z_CBPRINTF_SUPPRESS_SIZEOF_ARRAY_DECAY \
346 BUILD_ASSERT(!IS_ENABLED(CONFIG_XTENSA) || \
347 (IS_ENABLED(CONFIG_XTENSA) && \
348 !(_align_offset % CBPRINTF_PACKAGE_ALIGNMENT)), \
349 "Xtensa requires aligned package."); \
350 BUILD_ASSERT((_align_offset % sizeof(int)) == 0, \
351 "Alignment offset must be multiply of a word."); \
352 IF_ENABLED(CONFIG_CBPRINTF_STATIC_PACKAGE_CHECK_ALIGNMENT, \
353 (__ASSERT(!((uintptr_t)buf & (CBPRINTF_PACKAGE_ALIGNMENT - 1)), \
354 "Buffer must be aligned.");)) \
355 bool str_idxs = _flags & CBPRINTF_PACKAGE_ADD_STRING_IDXS; \
356 uint8_t *_pbuf = buf; \
357 uint8_t _s_cnt = 0; \
358 uint16_t _s_buffer[16]; \
359 size_t _pmax = (buf != NULL) ? _inlen : INT32_MAX; \
360 int _pkg_len = 0; \
361 int _total_len = 0; \
362 int _pkg_offset = _align_offset; \
363 union z_cbprintf_hdr *_len_loc; \
364 /* package starts with string address and field with length */ \
365 if (_pmax < sizeof(union z_cbprintf_hdr)) { \
366 _outlen = -ENOSPC; \
367 break; \
368 } \
369 _len_loc = (union z_cbprintf_hdr *)_pbuf; \
370 _pkg_len += sizeof(union z_cbprintf_hdr); \
371 _pkg_offset += sizeof(union z_cbprintf_hdr); \
372 /* Pack remaining arguments */\
373 FOR_EACH(Z_CBPRINTF_PACK_ARG, (;), __VA_ARGS__);\
374 _total_len = _pkg_len; \
375 if (str_idxs) {\
376 _total_len += _s_cnt; \
377 if (_pbuf) { \
378 for (int i = 0; i < _s_cnt; i++) { \
379 _pbuf[_pkg_len + i] = _s_buffer[i]; \
380 } \
381 } \
382 } \
383 /* Store length */ \
384 _outlen = (_total_len > (int)_pmax) ? -ENOSPC : _total_len; \
385 /* Store length in the header, set number of dumped strings to 0 */ \
386 if (_pbuf) { \
387 union z_cbprintf_hdr hdr = { \
388 .desc = { \
389 .len = (uint8_t)(_pkg_len / sizeof(int)), \
390 .str_cnt = 0, \
391 .ro_str_cnt = str_idxs ? _s_cnt : (uint8_t)0, \
392 } \
393 }; \
394 *_len_loc = hdr; \
395 } \
396 _Pragma("GCC diagnostic pop") \
397} while (0)
