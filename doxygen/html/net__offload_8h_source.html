34static inline int32_t timeout_to_int32(k_timeout_t timeout)
35{
36 if (K_TIMEOUT_EQ(timeout, K_NO_WAIT)) {
37 return 0;
38 } else if (K_TIMEOUT_EQ(timeout, K_FOREVER)) {
39 return -1;
40 } else {
41 return k_ticks_to_ms_floor32(timeout.ticks);
42 }
43}
