27/* Declare a reserved memory section */

30 _DT_RESERVED_START(res) = .; \

32 KEEP(*(._DT_LABEL_TOKEN(res).*)) \
33 _DT_RESERVED_END(res) = \

35 } GROUP_LINK_IN(_DT_LABEL_TOKEN(res))
