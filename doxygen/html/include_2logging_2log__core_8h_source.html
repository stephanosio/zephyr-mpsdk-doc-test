319#define Z_LOG2(_level, _inst, _source, _dsource, ...) do { \
320 if (!Z_LOG_CONST_LEVEL_CHECK(_level)) { \
321 break; \
322 } \
323 if (IS_ENABLED(CONFIG_LOG_MODE_MINIMAL)) { \
324 Z_LOG_TO_PRINTK(_level, __VA_ARGS__); \
325 break; \
326 } \
327 /* For instance logging check instance specific static level */ \
328 if (_inst & !IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING)) { \
329 if (_level > ((struct log_source_const_data *)_source)->level) { \
330 break; \
331 } \
332 } \
333 \
334 bool is_user_context = k_is_user_context(); \
335 uint32_t filters = IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING) ? \
336 (_dsource)->filters : 0;\
337 if (IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING) && !is_user_context && \
338 _level > Z_LOG_RUNTIME_FILTER(filters)) { \
339 break; \
340 } \
341 if (IS_ENABLED(CONFIG_LOG2)) { \
342 int _mode; \
343 void *_src = IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING) ? \
344 (void *)_dsource : (void *)_source; \
345 Z_LOG_MSG2_CREATE(UTIL_NOT(IS_ENABLED(CONFIG_USERSPACE)), _mode, \
346 CONFIG_LOG_DOMAIN_ID, _src, _level, NULL,\
347 0, __VA_ARGS__); \
348 } else { \
349 Z_LOG_INTERNAL(is_user_context, _level, \
350 _source, _dsource, __VA_ARGS__);\
351 } \
352 if (false) { \
353 /* Arguments checker present but never evaluated.*/ \
354 /* Placed here to ensure that __VA_ARGS__ are*/ \
355 /* evaluated once when log is enabled.*/ \
356 z_log_printf_arg_checker(__VA_ARGS__); \
357 } \
358} while (false)

396#define Z_LOG_HEXDUMP2(_level, _inst, _source, _dsource, _data, _len, ...) do { \
397 const char *_str = GET_ARG_N(1, __VA_ARGS__); \
398 if (!Z_LOG_CONST_LEVEL_CHECK(_level)) { \
399 break; \
400 } \
401 /* For instance logging check instance specific static level */ \
402 if (_inst & !IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING)) { \
403 if (_level > ((struct log_source_const_data *)_source)->level) { \
404 break; \
405 } \
406 } \
407 bool is_user_context = k_is_user_context(); \
408 uint32_t filters = IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING) ? \
409 (_dsource)->filters : 0;\
410 \
411 if (IS_ENABLED(CONFIG_LOG_MODE_MINIMAL)) { \
412 Z_LOG_TO_PRINTK(_level, "%s", _str); \
413 z_log_minimal_hexdump_print(_level, \
414 (const char *)_data, _len);\
415 break; \
416 } \
417 if (IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING) && !is_user_context && \
418 _level > Z_LOG_RUNTIME_FILTER(filters)) { \
419 break; \
420 } \
421 if (IS_ENABLED(CONFIG_LOG2)) { \
422 int mode; \
423 void *_src = IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING) ? \
424 (void *)_dsource : (void *)_source; \
425 Z_LOG_MSG2_CREATE(UTIL_NOT(IS_ENABLED(CONFIG_USERSPACE)), mode, \
426 CONFIG_LOG_DOMAIN_ID, _src, _level, \
427 _data, _len, \
428 COND_CODE_0(NUM_VA_ARGS_LESS_1(_, ##__VA_ARGS__), \
429 (), \
430 (COND_CODE_0(NUM_VA_ARGS_LESS_1(__VA_ARGS__), \
431 ("%s", __VA_ARGS__), (__VA_ARGS__)))));\
432 break; \
433 } \
434 uint16_t src_id = \
435 IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING) ? \
436 LOG_DYNAMIC_ID_GET(_dsource) : LOG_CONST_ID_GET(_source);\
437 struct log_msg_ids src_level = { \
438 .level = _level, \
439 .domain_id = CONFIG_LOG_DOMAIN_ID, \
440 .source_id = src_id, \
441 }; \
442 if (is_user_context) { \
443 log_hexdump_from_user(src_level, _str, \
444 (const char *)_data, _len); \
445 } else if (IS_ENABLED(CONFIG_LOG_MODE_IMMEDIATE)) { \
446 log_hexdump_sync(src_level, _str, (const char *)_data, _len); \
447 } else { \
448 log_hexdump(_str, (const char *)_data, _len, src_level); \
449 } \
450} while (false)

785#define __LOG_VA(_level, _source, _dsource, _str, _valist, _argnum, _strdup_action) do { \

797 if (IS_ENABLED(CONFIG_LOG_RUNTIME_FILTERING) && !is_user_context && \
798 _level > Z_LOG_RUNTIME_FILTER(filters)) { \
799 break; \
800 } \
801 if (IS_ENABLED(CONFIG_LOG2)) { \
802 z_log_msg2_runtime_vcreate(CONFIG_LOG_DOMAIN_ID, _source, \
